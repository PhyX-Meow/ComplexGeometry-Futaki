\begin{problem}
    Prove \[
        R(fX,gY)(hs)=fghR(X,Y)s
    \] for any vector fields \(X,Y\) and any section \(s\in C^\infty(M,E)\) and any
    smooth functions \(f,g,h\).
\end{problem}
\begin{problem}
    Prove
    \begin{enumerate}[(1)]
    \item \(R=\dd^\nabla\circ \dd^\nabla\).
    \item \(\dd^\nabla R=0\).
    \end{enumerate}
\end{problem}
